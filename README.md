- Create an account: https://supabase.com

- Create a new project

- Find **SQL Editor** (Or press CTRL + E in your Project Overview)

- Paste this code:
```
create table app_universal (
  id bigint generated by default as identity primary key,
  created_at timestamp with time zone default timezone('utc'::text, now()) not null,
  collection text not null,
  data jsonb not null,
  user_id uuid references auth.users default auth.uid()
);

alter table app_universal enable row level security;

create policy "Users can insert their own data" on app_universal for insert with check (auth.uid() = user_id);
create policy "Users can view their own data" on app_universal for select using (auth.uid() = user_id);
create policy "Users can update their own data" on app_universal for update using (auth.uid() = user_id);
create policy "Users can delete their own data" on app_universal for delete using (auth.uid() = user_id);
```
